83  RPF_ENABLE_MATERIAL = 1 << 5, // whether to enable material

85  RPF_CLEAR_COLOR = 1 << 6, // whether to clear the color buffer
86  RPF_CLEAR_DEPTH = 1 << 7, // whether to clear the depth buffer
87  RPF_CLEAR_STENCIL = 1 << 8, // whether to clear the depth buffer
88  RPF_CLEAR_ACCUM = 1 << 9, // whether to clear the accumulation buffer
89  RPF_CLEAR_ALL = RPF_CLEAR_COLOR|RPF_CLEAR_DEPTH|RPF_CLEAR_STENCIL|RPF_CLEAR_ACCUM, // whether to clear all buffers
90  RPF_DRAWABLES_INIT_FRAME = 1 << 10, // whether to call the init_frame method of the drawables
91  RPF_SET_STATE_FLAGS = 1 << 11, // whether to set depth buffer and culling flags
92  RPF_SET_CLEAR_COLOR = 1 << 12, // whether to set the clear color
93  RPF_SET_CLEAR_DEPTH = 1 << 13, // whether to set the clear color
94  RPF_SET_CLEAR_STENCIL = 1 << 14, // whether to set the clear color
95  RPF_SET_CLEAR_ACCUM = 1 << 15, // whether to set the accumulation buffer clear color
96  RPF_DRAWABLES_DRAW = 1 << 16, // whether to call draw and finish_draw methods of drawables
97  RPF_DRAWABLES_FINISH_FRAME = 1 << 17, // whether to call finish frame method of drawables
98  RPF_DRAW_TEXTUAL_INFO = 1 << 18, // whether to draw textual information
99  RPF_DRAWABLES_AFTER_FINISH = 1 << 19, // whether to call after finish method of drawables
100  RPF_HANDLE_SCREEN_SHOT = 1 << 20, // whether to perform a screen shot if this was scheduled
101  RPF_ALL = (1 << 21) - 1, // all flags set, defines default render pass
102  RPF_DEFAULT = RPF_ALL & ~ (RPF_SET_LIGHTS|RPF_CLEAR_ACCUM|RPF_SET_CLEAR_ACCUM|RPF_CLEAR_STENCIL|RPF_SET_CLEAR_STENCIL) // all flags set, defines default render pass
