VR support

cgv/libs/vr
	vr_event.h              // defines types to store the state of a vr kit
		vr::vr_trackable_state    // status[detached|attached|tracked]; pose:float[12]
		  vr::vr_controller_state // time_stamp; button_flags; axes[8]; vibration[2]
		vr::vr_kit_state          // hmd + controllers[2] (0=left, 1=right)

	vr_kit.h                // defines vr_kit class
		vr::vr_kit                // query_state, init_fbos, enable_fbos, submit_frame

	vr_driver.h             // defines vr_driver class and provides registry for drivers and kits
		vr::vr_driver             // installed, get_driver_name, scan_vr_kits, x|up_dir, action zone
        vr::scan_vr_kits()        // scan vr kits with all drivers
		vr::get_vr_kit(void*)     // resolve vr_kit pointer from handle

	gl_vr_display.h			// defines gl_vr_display class that implements offscreen rendering
		vr::gl_vr_display		  // opengl implementation of init_fbos, enable_fbos, ...
